* [[id:f23824a1-0515-47c6-b386-21d83a9aec21][PROBLEM]]

#+begin_src cpp
class Solution {
public:
  ListNode *reverseList(ListNode *head) {
    ListNode *nextNode = nullptr;
    ListNode *prevNode = nullptr;

    while (head) {
      nextNode = head->next;
      head->next = prevNode;
      prevNode = head;
      head = nextNode;
    }
    return prevNode;
  }
};
